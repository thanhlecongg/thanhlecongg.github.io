              if (restLen == 1) {

                  throw new IOException(
                      "Failed to read " + "Paxheader. Expected " + restLen + " bytes, read " + got);
                }
                
                final String value = new String(rest, 0, restLen - 1, CharsetNames.UTF_8);

              }
