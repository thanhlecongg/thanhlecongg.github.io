    long result = 0;
    int end = offset + length;
    int start = offset;

    }

    if (buffer[start] == 0) {
      return 0L;
    }

    
    while (start < end) {
      if (buffer[start] == ' ') {
        start++;
      } else {
        break;
      }
    }

    while (start < end && (trailer == 0 || trailer == ' ')) {
      end--;
      trailer = buffer[end - 1];
    }
    if (start == end) {
      throw new IllegalArgumentException(exceptionMessage(buffer, offset, length, start, trailer));
    }

    for (; start < end; start++) {
      final byte currentByte = buffer[start];

    }
